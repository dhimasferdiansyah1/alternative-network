// app/api/remove-transaction/route.ts

import { transactionStorage } from "../create-transaction/route";

export async function POST(request: Request) {
  const url = new URL(request.url);
  const orderId = url.searchParams.get("orderId");

  if (!orderId) {
    return new Response(
      JSON.stringify({
        error: "Invalid request",
        details: "Order ID is required",
      }),
      {
        status: 400,
        headers: { "Content-Type": "application/json" },
      }
    );
  }

  delete transactionStorage[orderId];

  return new Response(
    JSON.stringify({ success: true, message: "Transaction removed" }),
    {
      status: 200,
      headers: { "Content-Type": "application/json" },
    }
  );
}
